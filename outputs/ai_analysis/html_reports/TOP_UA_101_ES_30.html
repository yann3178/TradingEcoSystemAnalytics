	DaytoFilter(0);

// DEFINITIONS
SMA1_var = AverageFC(C, SMA1Length);

// CONDITIONS
if dayofweek(date) <> DaytoFilter and mycount < duration 
and PatternNeutralFast(PtnNeutYes, ohlcValues) and PatternNeutralFast(PtnNeutNo, ohlcValues)=false then begin

if SMA1_var crosses over SMA2_var and PatternDirectionalFast(+PtnDirYes, ohlcValues) 
	and PatternDirectionalFast(+PtnDirNo, ohlcValues)=False then buy("LE Cross") next bar market;
if SMA1_var crosses under SMA2_var and PatternDirectionalFast(-PtnDirYes, ohlcValues) 
	and PatternDirectionalFast(-PtnDirNo, ohlcValues)=False then sellshort("SE Cross") next bar market;

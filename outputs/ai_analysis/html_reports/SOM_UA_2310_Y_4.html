isStartOfSession = _OHLCMulti5(sessionStart, sessionEnd, ohlc);
  
superTrend = SuperTrendOliverSeban(atrLength, atrMult, ST);
switchLong = ST[1] < 0 and ST > 0;
switchShort = ST[1] > 0 and ST < 0;

if isStartOfSession and mp  <> 0
	then sessionsInTrade = sessionsInTrade + 1;
	
if mp <> mp[1] and mp <> 0 then sessionsInTrade = 1;
  
if tw(tradeStart, tradeEnd)
then begin
  if switchLong 
     and not PatternDirectionalFast(ptnDirN, ohlc) 
     and PatternFast(ptnLongY, ohlc)
     and not PatternFast(ptnLongN, ohlc)
  then buy("LE") myContracts contracts next bar H stop;
  
  if switchShort
     and not PatternDirectionalFast(-ptnDirN, ohlc) 
     and not PatternFast(ptnShortN, ohlc) 
     and dayofweek(d) <> skipSessionS
  then sellshort("SE") myContracts contracts next bar L stop;
end;

if maxSessionInTrade > 0 
	 and sessionsInTrade >= maxSessionInTrade 
then
	setexitonclose;

setstopcontract;
if stopLoss > 0 then setstoploss(stopLoss);
if takeProfit > 0 then setprofittarget(takeProfit);
if breakEven > 0 then setbreakeven(breakEven);

if marketposition=1 then sell("LX_trail") next bar entryprice + DistOpenRunup*(1-TrailPercentDistance/100) stop;

input: mycontracts(1);
inputs: Length(20), NumATRs(3), DayFilter(4), SL(3500), TP(6500);

variables: Avg(0), Shift(0), UpperBand(0), LowerBand(0);

Avg = AverageFC(close, Length);
Shift = NumATRs * AvgTrueRange(Length);
UpperBand = Avg + Shift;
LowerBand = Avg - Shift;


if close crosses over UpperBand and dayofweek(d) <> DayFilter then buy mycontracts contracts next bar high stop;
if close crosses under LowerBand then sell next bar low stop;

setstopcontract;
setstoploss(SL);
if TP <> 0 then setprofittarget(TP);

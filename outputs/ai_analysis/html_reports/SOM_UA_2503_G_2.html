For conta = 1 to NbarLong begin
    SumBodyL = SumBodyL + AbsValue(Close[conta] - Open[conta]);
end;
AvgBodyL = SumBodyL / NbarLong;

// Calcolo del body medio delle precedenti N barre per Short
SumBodyS = 0;
For conta = 1 to NbarShort begin
    SumBodyS = SumBodyS + AbsValue(Close[conta] - Open[conta]);
end;
AvgBodyS = SumBodyS / NbarShort;


// Trigger: il body della barra corrente deve essere superiore a BodyFactor * AvgBody
TriggerL = (Close - Open) > BodyFactorLong * AvgBodyL;

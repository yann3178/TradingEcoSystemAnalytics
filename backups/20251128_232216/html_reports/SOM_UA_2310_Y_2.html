//DEFINITIONS

PercTP = (MyCapital*MyProfit)/100;

//CONDITIONS
If dayofweek(date) <> DaytoFilter and tw(MyStartTrade,MyEndTrade) then begin
	
	If PatternFast(MyPtnLY, ohlcValues) and PatternFast(MyPtnLN, ohlcValues)=False
	and SMA1var Crosses Over SMA2var then buy("LE_Cross" ) MySize contracts next bar at market;
	
	If PatternFast(MyPtnSY, ohlcValues) and PatternFast(MyPtnSN, ohlcValues)=False 
	and SMA1var Crosses Under SMA2var then sellshort("SE_Cross") MySize contracts next bar at market;

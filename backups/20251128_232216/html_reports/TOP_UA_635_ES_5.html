//Setting of Pivot Points Levels 

If isStartOfSession then begin
	MyPP = (HighS1+LowS1+CloseS1)/3;
	MyR1 = 2*MyPP - LowS1;
	MyS1 = 2*MyPP - HighS1;
     //MyR2 = MyPP + (HighD1-LowD1);

//Setting of Entry Setup Choices

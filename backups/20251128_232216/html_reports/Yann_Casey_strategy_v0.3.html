// Reset si timeout (eviter blocage)
if BounceState > 0 and CurrentBar > (BounceDetectedBar + 10) then begin
    BounceState = 0;
    BounceConfirmCount = 0;
end;

// ==================== CONDITIONS D'ENTREE AVEC ZONES CASEY ====================

// Conditions SHORT
CanEnterShort = (
    useShort = 1 and tw and
    SequenceActive and RetestDone and
    (BrokenZoneType = 2 or BrokenZoneType = 4 or BrokenZoneType = 5) and
    IsRegularSession and
    Close < EMA13 and EMA13 < EMA48 and EMA48 < EMA200 and
    EMA13 < BrokenZoneBottom and
    TradesCountToday < MaxTradesPerDay and
    MarketPosition = 0 and
    PDL_ZoneBottom > 0 and PDH_ZoneTop > 0 and
    Close < PDH_ZoneBottom and
    DynamicPositionSize > 0
);

// Conditions LONG
CanEnterLong = (
    useLong = 1 and tw and
    SequenceActive and RetestDone and
    (BrokenZoneType = 1 or BrokenZoneType = 3 or BrokenZoneType = 6) and
    IsRegularSession and
    Close > EMA13 and EMA13 > EMA48 and EMA48 > EMA200 and
    EMA13 > BrokenZoneBottom and
    TradesCountToday < MaxTradesPerDay and
    MarketPosition = 0 and
    PDL_ZoneBottom > 0 and PDH_ZoneTop > 0 and
    Close > PDL_ZoneTop and
    DynamicPositionSize > 0
);

// ==================== CALCUL DES TARGETS OPTIMISE AVEC TABLEAUX TRIES ====================

// Calcul target pour SHORT (prochaine zone de support Casey)
if CanEnterShort then begin
    TargetZoneTop = 0;
    TargetZoneBottom = 0;
    
    // Chercher premiere zone de support au-dessous (tableau trie ascendant)
    // Parcourir de la fin vers le debut pour trouver le support le plus proche
    for idx2 = SupportZonesCount - 1 downto 0 begin
        if SupportZoneStrength[idx2] > 0 and SupportZoneTop[idx2] < BrokenZoneBottom then begin
            TargetZoneTop = SupportZoneTop[idx2];
            TargetZoneBottom = SupportZoneBottom[idx2];
            break; // Premier trouve = le plus proche
        end;
    end;
end;

// Calcul target pour LONG (prochaine zone de resistance Casey)
if CanEnterLong then begin
    TargetZoneTop = 0;
    TargetZoneBottom = 0;
    
    // Chercher premiere zone de resistance au-dessus (tableau trie descendant)
    // Parcourir de la fin vers le debut pour trouver la resistance la plus proche
    for idx2 = ResistanceZonesCount - 1 downto 0 begin
        if ResistanceZoneStrength[idx2] > 0 and ResistanceZoneBottom[idx2] > BrokenZoneTop then begin
            TargetZoneTop = ResistanceZoneTop[idx2];
            TargetZoneBottom = ResistanceZoneBottom[idx2];
            break; // Premier trouve = le plus proche
        end;
    end;
end;

// ==================== EXECUTION DES ORDRES AVEC POSITION SIZING CASEY ====================

// ENTREE SHORT
if CanEnterShort then begin
    SellShort DynamicPositionSize Contract next bar at market;
    EntryZoneTop = BrokenZoneTop;
    EntryZoneBottom = BrokenZoneBottom;
    StopLevel = BrokenZoneTop + EpsilonSupport;
    
    TradesCountToday = TradesCountToday + 1;
    
    // Reset sequence
    SequenceActive = false;
    RetestDone = false;
    ZoneBroken = false;
    BounceState = 0;
end;

// ENTREE LONG
if CanEnterLong then begin
    Buy DynamicPositionSize Contract next bar at market;
    EntryZoneTop = BrokenZoneTop;
    EntryZoneBottom = BrokenZoneBottom;
    StopLevel = BrokenZoneBottom - EpsilonSupport;
    
    TradesCountToday = TradesCountToday + 1;
    
    // Reset sequence
    SequenceActive = false;
    RetestDone = false;
    ZoneBroken = false;
    BounceState = 0;
end;

// ==================== GESTION DES SORTIES VERS ZONES CASEY ====================

// Take Profit vers zones Casey
if MarketPosition = 1 and TargetZoneBottom > 0 then begin
    // Sortie partielle a l'entree de la zone target Casey
    if CurrentContracts >= 2 then
        Sell 1 contract next bar at TargetZoneBottom limit;
end;

if MarketPosition = -1 and TargetZoneTop > 0 then begin
    // Sortie partielle a l'entree de la zone target Casey
    if CurrentContracts >= 2 then
        BuyToCover 1 contract next bar at TargetZoneTop limit;
end;

// Stop Loss sur zone d'entree Casey
if MarketPosition = 1 and StopLevel > 0 then begin
    Sell next bar at StopLevel stop;
end;

if MarketPosition = -1 and StopLevel > 0 then begin
    BuyToCover next bar at StopLevel stop;
end;

// Trailing Stop EMA13 (Casey-style)
if MarketPosition = 1 and Close < (EMA13 - EpsilonSupport) then begin
    Sell next bar at market;
end;

if MarketPosition = -1 and Close > (EMA13 + EpsilonSupport) then begin
    BuyToCover next bar at market;
end;

// Sortie fin de journee
if IsRegularSession and Time >= (RegularSessionEnd - 10) then begin
    if MarketPosition = 1 then Sell next bar at market;
    if MarketPosition = -1 then BuyToCover next bar at market;
end;

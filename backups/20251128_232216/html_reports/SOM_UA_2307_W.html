var: IsStartOfSession(true), highd1(0), lowd1(0); //here we use as an example high and low of previous sessions
array: ohlcValues[23](0);

inputs: sessionStartTimeC(1800),sessionEndTimeC(1700);
Input: randomValue(0);
IsStartOfSession = _OHLCMulti5Rev2(sessionStartTimeC, sessionEndTimeC, ohlcValues, randomValue);
highd1 = ohlcvalues[5]; lowd1 = ohlcvalues[6]; //these are picked depending on the chosen trigger, in this example high and low of last session

//Works using 60-minute bars but can also work down to 5 minutes bars

Input: PtnNeutYes(33), PtnNeutNo(27), PtnDirYes(1), PtnDirNo(-25);
Input: MyStartTrade(500), MyEndTrade(1430), MyStartPause(1200), MyEndPause(1300);
Input: MyStop(1500), MyProfit(0);
Input: MyContracts(1);

//here below we use patterns in a mirrorerd setup so that the same or opposite condition appliy to long or short
// to do so either neutral patterns or mirrored directional patterns are used
If Time > MyStartTrade And time < MyEndTrade and (Time < MyStartPause or time > MyEndPause) //time window
and PatternNeutralFast(PtnNeutYes, ohlcValues) and PatternNeutralFast(PtnNeutNo, ohlcValues)=false then begin

	If PatternDirectionalFast(+PtnDirYes, ohlcValues) and PatternDirectionalFast(+PtnDirNo, ohlcValues)=False 
	then buy MyContracts contracts next bar at highd1 stop;
	
	If PatternDirectionalFast(-PtnDirYes, ohlcValues) and PatternDirectionalFast(-PtnDirNo, ohlcValues)=False 
	then sellshort MyContracts contracts next bar at lowd1 stop;

end;

setstopcontract;

If MyStop > 0 then setstoploss(MyStop);
If MyProfit > 0 then setprofittarget(MyProfit);

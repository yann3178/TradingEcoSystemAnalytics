//	Template: _UA_Mirrored_TF
//  	Instrument: Koffe (@KC)

//  	In-Sample: 01-01-2010 / 31-12-2020

Input: Size(1), SessionStart(0415), SessionEnd(1330),
	StartTimeH(0600), StartTimeM(45), EndTimeH(800), EndTimeM(0),
	PtnNeutY(4), PtnNeutN(47),
	PtnDirY(28), PtnDirN(11),
	CloseTimeH(1300), CloseTimeM(15),
	StopLoss(1700), TakeProfit(2500);

array: ohlcValues[23](0);
vars: startTime(0), endTime(0), highd1 (0), lowd1 (0), IsStartOfSession(false);

IsStartOfSession = _OHLCMulti5(SessionStart, SessionEnd, ohlcValues);
highd1 = ohlcvalues[5]; lowd1 = ohlcvalues[6];
startTime = StartTimeH + StartTimeM;
endTime = EndTimeH + EndTimeM;

If tw(startTime, endTime)
And PatternNeutralFast(PtnNeutY, ohlcValues) 
And PatternNeutralFast(PtnNeutN, ohlcValues) = false then begin

	If PatternDirectionalFast(+PtnDirY, ohlcValues) 
	And PatternDirectionalFast(+PtnDirN, ohlcValues) = False then 
		Buy ("LE") Size contracts next bar at highd1 stop;
	
	If PatternDirectionalFast(-PtnDirY, ohlcValues) 
	And PatternDirectionalFast(-PtnDirN, ohlcValues) = False then
		SellShort ("SE") Size contracts next bar at lowd1 stop;

end;

// Close at end of session

	If marketposition = 1 then Sell ("EOS_LX") Size Contracts next bar at market;
	If marketposition = -1 then BuyToCover ("EOS_SX") Size Contracts next bar at market;
end;

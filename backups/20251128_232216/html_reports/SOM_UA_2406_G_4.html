input:MyContracts(1),
	sessionStart(0800),
	sessionEnd(2200),
	tradeStart(0830),
	tradeEnd(1600),
	stopLossL(1300),
	stopLossS(1250),
	takeProfitL(6500),
	takeProfitS(7500),
	ptnLY(41),
	ptnLN(19),
	ptnSY(11),
	ptnSN(37),
	adxTh(30),
	skipL(-1),
	skipS(3);
	
var:
	isStartOfSession(false),
	adxVal(0),
	highS1(0),
	lowS1(0);
	
arrays: 
	ohlc[23](0),
	adxValues[3](0);

isStartOfSession = _OHLCMulti5(sessionStart, sessionEnd, ohlc);
highS1 = ohlc[5];
lowS1 = ohlc[6];

if isStartOfSession then begin
	adxVal = iADXOnArray(5, ohlc[5], ohlc[6], ohlc[7], ohlc[9], ohlc[10], ohlc[11], adxValues) * 100;
end;

if tw(tradeStart, tradeEnd)
 	 and EntriesToday(d) = 0
 	 and adxVal < adxTh
then begin
	if dayofweek(d) <> skipL
	   and UAPtnBase(ptnLY, ohlc)
	   and not UAPtnBase(ptnLN, ohlc)
	then buy("LE") MyContracts contracts next bar at highS1 stop;
	
	if dayofweek(d) <> skipS
	   and UAPtnBase(ptnSY, ohlc)
	   and not UAPtnBase(ptnSN, ohlc)	
	then sellshort("SE") MyContracts contracts next bar at lowS1 stop;
end;

setexitonclose;
setstopcontract;

if stopLossL > 0 or stopLossS > 0 then setstoploss(maxlist(stopLossL, stopLossS));
if marketposition =  1 and stopLossL > 0 then setstoploss(stopLossL);
if marketposition = -1 and stopLossS > 0 then setstoploss(stopLossS);

if takeProfitL > 0 or takeProfitS > 0 then setprofittarget(maxlist(takeProfitL, takeProfitS));
if marketposition =  1 and takeProfitL > 0 then setprofittarget(takeProfitL);
if marketposition = -1 and takeProfitS > 0 then setprofittarget(takeProfitS);

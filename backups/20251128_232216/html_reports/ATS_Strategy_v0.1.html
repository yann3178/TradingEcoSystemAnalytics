    epsilon_cluster = 0.01; {1% de distance}
    
    {D?tection du cluster de support}
    GF_isSupportCluster = false;
    SupportClusterCount = 0;
    
    For sort_i = 1 to supportIdx Begin
        SupportClusterCount = 0;
        
        For sort_j = 1 to supportIdx Begin
            {Si l'?l?ment j est ? une distance epsilon de l'?l?ment i}
            If sort_i <> sort_j AND absvalue(GF_Supports[sort_i] - GF_Supports[sort_j]) / GF_Supports[sort_i] <= epsilon_cluster Then
                SupportClusterCount = SupportClusterCount + 1;
        End;
        
        {Si on a au moins 2 autres supports proches (donc 3 au total), c'est un cluster}
        If SupportClusterCount >= 2 AND GF_Supports[i] = GF_ClosestSupport Then
            GF_isSupportCluster = true;
    End;
    
    {D?tection du cluster de r?sistance}
    GF_isResistanceCluster = false;
    ResistanceClusterCount = 0;	
    
    For sort_i = 1 to resistanceIdx Begin
        ResistanceClusterCount = 0;
        
        For sort_j = 1 to resistanceIdx Begin
            {Si l'?l?ment j est ? une distance epsilon de l'?l?ment i}
            If sort_i <> sort_j AND absvalue(GF_Resistances[sort_i] - GF_Resistances[sort_j]) / GF_Resistances[sort_i] <= epsilon_cluster Then
                ResistanceClusterCount = ResistanceClusterCount + 1;
        End;
        
        {Si on a au moins 2 autres r?sistances proches (donc 3 au total), c'est un cluster}
        If ResistanceClusterCount >= 2 AND GF_Resistances[i] = GF_ClosestResistance Then
            GF_isResistanceCluster = true;
    End;
    
    {Afficher les r?sultats de la Grande Fen?tre - pour d?bogage}
    Print("GF ClosestSupport: ", GF_ClosestSupport, " (", GF_SupportName, ") - Cluster: ", GF_isSupportCluster);
    Print("GF ClosestResistance: ", GF_ClosestResistance, " (", GF_ResistanceName, ") - Cluster: ", GF_isResistanceCluster);

End;

// ===== TREND SCALING ALGORITHM FOR NQ =====
// Advanced trend following with adaptive position sizing and reversal detection

Inputs:
    // Trend Detection
    FastSMA(50),
    SlowSMA(250),
    RSIPeriod(14),
    ROCPeriod(10),
    
    // Position Scaling
    BaseContracts(1),
    MaxContracts(5),
    ScaleThreshold(0.5), // ATR multiplier for scaling
    
    // Risk Management
    InitialStop(150),
    TrailingATRMult(2.0),
    MaxDailyRisk(2000),
    
    // Reversal Detection
    VIXThreshold(25),
    RSIOverbought(75),
    RSIOversold(25),
    ROCExtreme(5),
    
    // Time Management
    SessionStart(930),
    SessionEnd(1600),
    NoTradeStart(1130),
    NoTradeEnd(1300);

Variables:
    // Trend Analysis
    FastMA(0), SlowMA(0), 
    RSIVal(0), ROCVal(0),
    TrendStrength(0), TrendDirection(0),
    
    // Position Management
    CurrentContracts(0),
    EntryPrice(0),
    LastScalePrice(0),
    DailyPnL(0),
    MaxDailyLoss(0),
    
    // Volatility & Risk
    ATRVal(0),
    VIXLevel(0),
    VolatilityRegime(0), // 1=Low, 2=Medium, 3=High
    
    // Signals
    LongEntry(False),
    ShortEntry(False),
    ScaleLong(False),
    ScaleShort(False),
    TakeProfit(False),
    StopLoss(False),
    
    // Counters
    BarsSinceEntry(0),
    ConsecutiveLosses(0),
    
    // Session Management
    IsNewSession(False),
    IsTradeTime(False);

// ===== INITIALIZATION =====
if Date <> Date[1] then begin
    DailyPnL = 0;
    MaxDailyLoss = -MaxDailyRisk;
    IsNewSession = True;
end else
    IsNewSession = False;

// ===== TIME MANAGEMENT =====
IsTradeTime = Time >= SessionStart and Time <= SessionEnd and 
              not (Time >= NoTradeStart and Time <= NoTradeEnd);

// ===== TECHNICAL INDICATORS =====
FastMA = Average(Close, FastSMA);
SlowMA = Average(Close, SlowSMA);
RSIVal = RSI(Close, RSIPeriod);
ROCVal = RateOfChange(Close, ROCPeriod);
ATRVal = AvgTrueRange(20);

// Simulated VIX proxy using NQ volatility
VIXLevel = (ATRVal / Close) * 100 * 10; // Approximate VIX equivalent

// ===== VOLATILITY REGIME DETECTION =====
if VIXLevel < 20 then
    VolatilityRegime = 1 // Low vol
else if VIXLevel < 30 then
    VolatilityRegime = 2 // Medium vol
else
    VolatilityRegime = 3; // High vol

// ===== TREND ANALYSIS =====
if FastMA > SlowMA then begin
    TrendDirection = 1; // Bullish
    TrendStrength = (FastMA - SlowMA) / ATRVal;
end else begin
    TrendDirection = -1; // Bearish
    TrendStrength = (SlowMA - FastMA) / ATRVal;
end;

// ===== ENTRY SIGNALS =====
// Long Entry Conditions
LongEntry = IsTradeTime and 
            TrendDirection = 1 and
            Close crosses above FastMA and
            FastMA > SlowMA and
            RSIVal > 50 and RSIVal < RSIOverbought and
            ROCVal > 0 and
            VIXLevel < VIXThreshold and
            DailyPnL > MaxDailyLoss and
            MarketPosition = 0;

// Short Entry Conditions  
ShortEntry = IsTradeTime and
             TrendDirection = -1 and
             Close crosses below FastMA and
             FastMA < SlowMA and
             RSIVal < 50 and RSIVal > RSIOversold and
             ROCVal < 0 and
             VIXLevel < VIXThreshold and
             DailyPnL > MaxDailyLoss and
             MarketPosition = 0;

// ===== SCALING LOGIC =====
if MarketPosition = 1 then begin // Long position
    ScaleLong = Close > EntryPrice + (ScaleThreshold * ATRVal) and
                CurrentContracts < MaxContracts and
                TrendStrength > 2 and
                RSIVal < RSIOverbought and
                BarsSinceEntry >= 3;
end;

if MarketPosition = -1 then begin // Short position
    ScaleShort = Close < EntryPrice - (ScaleThreshold * ATRVal) and
                 CurrentContracts < MaxContracts and
                 TrendStrength > 2 and
                 RSIVal > RSIOversold and
                 BarsSinceEntry >= 3;
end;

// ===== EXIT LOGIC =====

// Trend Weakening Exit
if MarketPosition = 1 and (TrendStrength < 1 or RSIVal > RSIOverbought) then
    TakeProfit = True;

if MarketPosition = -1 and (TrendStrength < 1 or RSIVal < RSIOversold) then
    TakeProfit = True;

// Reversal Detection Exit
if MarketPosition = 1 and (FastMA crosses below SlowMA or VIXLevel > VIXThreshold * 1.2) then
    StopLoss = True;

if MarketPosition = -1 and (FastMA crosses above SlowMA or VIXLevel > VIXThreshold * 1.2) then
    StopLoss = True;

// Time-based Exit
if Time >= SessionEnd then begin
    if MarketPosition = 1 then
        Sell next bar at market;
    if MarketPosition = -1 then
        BuyToCover next bar at market;
end;

// ===== ORDER EXECUTION =====

// Initial Entries
if LongEntry then begin
    Buy BaseContracts contracts next bar at market;
    EntryPrice = Close;
    CurrentContracts = BaseContracts;
    LastScalePrice = Close;
    BarsSinceEntry = 0;
end;

if ShortEntry then begin
    SellShort BaseContracts contracts next bar at market;
    EntryPrice = Close;
    CurrentContracts = BaseContracts;
    LastScalePrice = Close;
    BarsSinceEntry = 0;
end;

if ScaleLong then begin
    Buy 1 contract next bar at market;
    CurrentContracts = CurrentContracts + 1;
    LastScalePrice = Close;

    SellShort 1 contract next bar at market;
    CurrentContracts = CurrentContracts + 1;

    if MarketPosition = -1 and CurrentContracts > 1 then
        BuyToCover (CurrentContracts / 2) contracts next bar at market;
end;

// Stop Loss (Full Exit)
if StopLoss then begin
    if MarketPosition = 1 then
        Sell next bar at market;
    if MarketPosition = -1 then
        BuyToCover next bar at market;
end;

// ===== RISK MANAGEMENT =====
SetStopContract;

// Adaptive Stop Loss
if MarketPosition <> 0 then begin
    if VolatilityRegime = 1 then
        SetStopLoss(InitialStop * 0.8)
    else if VolatilityRegime = 3 then
        SetStopLoss(InitialStop * 1.5)
    else
        SetStopLoss(InitialStop);
        
    // Trailing Stop
    SetDollarTrailing(TrailingATRMult * ATRVal * BigPointValue);
end;

if MarketPosition <> 0 then
    BarsSinceEntry = BarsSinceEntry + 1;

if MarketPosition = 0 then begin
    CurrentContracts = 0;
    BarsSinceEntry = 0;
end;

// ===== PERFORMANCE MONITORING =====
DailyPnL = NetProfit - NetProfit[1];

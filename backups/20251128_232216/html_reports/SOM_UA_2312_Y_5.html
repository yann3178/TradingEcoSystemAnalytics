üíª Strategy Code

Input: MyCapital(10000), StopLoss(4.5), TakeProfit(6.5);
Input: StartH(1100), StartM(0), EndH(1500), EndM(45), DF(0.8);
var: MySize(0), ranged1(0);

MySize = MyCapital/c;

ranged1 = highS(1)- lowS(1);

if ranged1 <> 0 then
condition1 = absvalue (closeS(1)-openS(1)) / ranged1 <= DF;

if condition1 and t>= (StartH + StartM) and t < (EndH + EndM) then begin
        if TradesToday(d)<1 then buy("LE") MySize contracts next bar HighS(0) stop;
end;

if t>= (StartH + StartM) then sell("SE") next bar LowS(0) stop;

setstopposition;

if StopLoss>0 then setstoploss(MyCapital*stoploss/100);
if TakeProfit>0 then setprofittarget(MyCapital*takeprofit/100);

    If GF_ScenarioLongRenforcePar <> "" Then
        Print("Renforce par: ", GF_ScenarioLongRenforcePar);
End;

If GF_CodeScenarioShort <> "" Then Begin
    Print("Scenario Baissier Detecte: ", GF_CodeScenarioShort);
    Print(GF_PossibleScenarioShort);
    If GF_ScenarioShortRenforcePar <> "" Then
        Print("Renforce par: ", GF_ScenarioShortRenforcePar);
End;

Print("DEBUG FIN ANALYSE: GF_CodeScenarioLong=", GF_CodeScenarioLong);

{===========================================================}
{PARTIE 2: TRADING - Executee a chaque barre}

{Variables de suivi des positions et trading optimisees - OPTIMISE}
Vars:
    slHit(false), tpHit(false), hasRecentSignal(false), ut(0);

{Journalisation des signaux - seulement aux changements de jour}
If Date <> Date[1] Then Begin
    Print("ETAT DU JOUR: Date=", Date, 
          " GF_CodeScenarioLong=", GF_CodeScenarioLong,
          " positionOpen=", positionOpen);
    
    {Journalisation des signaux sur tous les timeframes}
    For ut = TF_15MIN to TF_1HOUR Begin
        Print("SIGNAL: UT=", UTNames[ut], 
              " Code=", UTLastSignalCode[ut], 
              " Distance=", UTLastSignalDistance[ut]);
    End;
End;

{===========================================================}
{          Gestion d'une position ouverte existante         }
{===========================================================}
If positionOpen Then Begin
    {Calcul du profit actuel}
    If positionType = 1 Then {Position Long}
        currentProfit = (currentPrice - entryPrice) * BigPointValue
    Else {Position Short}
        currentProfit = (entryPrice - currentPrice) * BigPointValue;
    
    {Calcul simplifie du nombre de jours en position}
    If entryDate > 0 Then
        daysSinceEntry = IntPortion((currentDate - entryDate) / 10000)
    Else
        daysSinceEntry = 0;
    
    {Verification des conditions de sortie}
    shouldExit = false;
    
    {1. La duree maximale de la position est atteinte}
    If daysSinceEntry >= param_MaxDaysInTrade Then Begin
        shouldExit = true;
        Print("Duree maximale atteinte (", daysSinceEntry, " jours) - Sortie forcee");
    End;
    
    {2. Fin de session de trading et sortie automatique activee}
    If param_ExitOnSessionClose AND currentTime >= session_EndTime Then Begin
        shouldExit = true;
        Print("Fin de session (", currentTime, ") - Sortie forcee");
    End;
    
    {3. Vendredi et fermeture automatique activee}
    If param_CloseOnFriday AND isFriday AND currentTime >= 1600 Then Begin
        shouldExit = true;
        Print("Vendredi apres 16h - Sortie forcee");
    End;
    
    {4. Gestion du Break Even si active}
    If param_BEEnabled AND NOT breakEvenActive Then Begin
        {Verifier si le profit atteint le pourcentage d'activation}
        If positionType = 1 Then Begin {Position Long}
            If currentProfit >= (param_TPLong * param_BEActivationPct / 100) Then Begin
                stopPrice = entryPrice; {Remonter le SL au prix d'entree}
                breakEvenActive = true;
                Print("Break Even active pour position LONG a ", stopPrice);
            End;
        End
        Else If positionType = -1 Then Begin {Position Short}
            If currentProfit >= (param_TPShort * param_BEActivationPct / 100) Then Begin
                stopPrice = entryPrice; {Descendre le SL au prix d'entree}
                breakEvenActive = true;
                Print("Break Even active pour position SHORT a ", stopPrice);
            End;
        End;
    End;
    
    {5. Verification du Stop Loss et Take Profit}
    slHit = false;
    tpHit = false;
    
    If positionType = 1 Then Begin {Position Long}
        {Stop Loss - verifier si le prix est descendu en dessous du SL}
        If Low <= stopPrice Then Begin
            slHit = true;
            Print("Stop Loss atteint pour position LONG: SL=", stopPrice, " Low=", Low);
        End;
        
        {Take Profit - verifier si le prix est monte au-dessus du TP}
        If High >= targetPrice Then Begin
            tpHit = true;
            Print("Take Profit atteint pour position LONG: TP=", targetPrice, " High=", High);
        End;
    End
    Else If positionType = -1 Then Begin {Position Short}
        {Stop Loss - verifier si le prix est monte au-dessus du SL}
        If High >= stopPrice Then Begin
            slHit = true;
            Print("Stop Loss atteint pour position SHORT: SL=", stopPrice, " High=", High);
        End;
        
        {Take Profit - verifier si le prix est descendu en dessous du TP}
        If Low <= targetPrice Then Begin
            tpHit = true;
            Print("Take Profit atteint pour position SHORT: TP=", targetPrice, " Low=", Low);
        End;
    End;
    
    {Si SL ou TP atteint, sortir de la position}
    If slHit OR tpHit Then
        shouldExit = true;
    
    {Sortie de position si necessaire}
    If shouldExit Then Begin
        If positionType = 1 Then Begin {Position Long}
            Sell("Exit") myContract contract next bar at market;
            Print("Sortie de position LONG: Prix d'entree=", entryPrice, 
                " Prix actuel=", currentPrice, " P/L=", currentProfit);
        End
        Else If positionType = -1 Then Begin {Position Short}
            BuyToCover("Exit") myContract contract next bar at market;
            Print("Sortie de position SHORT: Prix d'entree=", entryPrice, 
                " Prix actuel=", currentPrice, " P/L=", currentProfit);
        End;
        
        {Reinitialiser les variables de position - CRITIQUE pour permettre de nouvelles entrees}
        Print("REINITIALISATION DES VARIABLES DE POSITION");
        positionOpen = false;
        positionType = 0;
        entryPrice = 0;
        stopPrice = 0;
        targetPrice = 0;
        entryDate = 0;
        entryTime = 0;
        breakEvenActive = false;
        currentProfit = 0;
    End;
End
{===========================================================}
{             evaluation des entrees en position            }
{===========================================================}
Else Begin
    {CONDITIONS CONSIDERABLEMENT ASSOUPLIES POUR LE TEST}
    {Verification des conditions d'entree en Long}
    If canTradeLong AND 
       GF_CodeScenarioLong <> "" Then Begin
        
        {ASSOUPLISSEMENT MAJEUR: tout signal haussier recent est accepte}
        hasRecentSignal = false;
        
        {Verification si un signal haussier recent existe sur n'importe quel timeframe}
        For ut = TF_15MIN to TF_1HOUR Begin
            For signalIdx = 1 to 5 Begin
                If UTLastSignalCode[ut] = signalHaussier[signalIdx] AND UTLastSignalDistance[ut] <= 5 Then
                    hasRecentSignal = true;
            End;
        End;
        
        If hasRecentSignal Then Begin
            {ACCEPTER TOUTE CONFIGURATION DE SUPPORT/RESISTANCE}
            canTradeLong = true;
            Print("Signal LONG valide (criteres assouplis) - GF_CodeScenarioLong=", GF_CodeScenarioLong);
        End;
    End;
    
    {Verification des conditions d'entree en Short - egalement assouplies}
    If canTradeShort AND 
       GF_CodeScenarioShort <> "" Then Begin
               
        {Verification si un signal baissier recent existe sur n'importe quel timeframe}
        For ut = TF_15MIN to TF_1HOUR Begin
            For signalIdx = 1 to 5 Begin
                If UTLastSignalCode[ut] = signalBaissier[signalIdx] AND UTLastSignalDistance[ut] <= 5 Then
                    hasRecentSignal = true;
            End;
        End;
        
        If hasRecentSignal Then Begin
            {ACCEPTER TOUTE CONFIGURATION DE SUPPORT/RESISTANCE}
            canTradeShort = true;
            Print("Signal SHORT valide (criteres assouplis) - GF_CodeScenarioShort=", GF_CodeScenarioShort);
        End;
    End;
    
    {AJOUTER UN CODE D'URGENCE POUR FORCER DES TRADES A DES MOMENTS SPECIFIQUES}
    If NOT canTradeLong AND NOT canTradeShort Then Begin
        {Force un trade long tous les lundis a 10h00}
        If DayOfWeek(currentDate) = 1 AND currentTime = 1000 Then Begin
            canTradeLong = true;
            Print("TRADE LONG FORCE - LUNDI 10H");
        End;
        
        {Force un trade short tous les mercredis a 15h00}
        If DayOfWeek(currentDate) = 3 AND currentTime = 1500 Then Begin
            canTradeShort = true;
            Print("TRADE SHORT FORCE - MERCREDI 15H");
        End;
    End;
    
    {Entree en position Long si les conditions sont remplies}
    If canTradeLong Then Begin
        Buy("EntryLong") myContract contract next bar at market;
        
        {Memorisation des donnees de position}
        positionOpen = true;
        positionType = 1;
        entryPrice = currentPrice;
        stopPrice = currentPrice - (param_SLLong / BigPointValue);
        targetPrice = currentPrice + (param_TPLong / BigPointValue);
        entryDate = currentDate;
        entryTime = currentTime;
        breakEvenActive = false;
        
        Print("Entree en position LONG: Prix=", entryPrice, 
            " SL=", stopPrice, " TP=", targetPrice,
            " Scenario=", GF_CodeScenarioLong);
    End
    {Entree en position Short si les conditions sont remplies}
    Else If canTradeShort Then Begin
        SellShort("EntryShort") myContract contract next bar at market;
        
        {Memorisation des donnees de position}
        positionOpen = true;
        positionType = -1;
        entryPrice = currentPrice;
        stopPrice = currentPrice + (param_SLShort / BigPointValue);
        targetPrice = currentPrice - (param_TPShort / BigPointValue);
        entryDate = currentDate;
        entryTime = currentTime;
        breakEvenActive = false;
        
        Print("Entree en position SHORT: Prix=", entryPrice, 
            " SL=", stopPrice, " TP=", targetPrice,
            " Scenario=", GF_CodeScenarioShort);
    End;
End;
